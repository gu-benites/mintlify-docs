export const DownloadButton = ({ platform, variant = "default", size = "md" }) => {
  const download = downloadLinks[platform];
  if (!download) return null;

  const sizeStyles = {
    sm: "px-4 py-2 text-sm",
    md: "px-6 py-3 text-base",
    lg: "px-8 py-4 text-lg"
  };

  const variantStyles = {
    default: "bg-slate-900 hover:bg-slate-800 text-white border-slate-900",
    outline: "bg-transparent hover:bg-slate-50 text-slate-900 border-slate-300 dark:text-slate-100 dark:border-slate-600 dark:hover:bg-slate-800",
    primary: "bg-green-600 hover:bg-green-700 text-white border-green-600"
  };

  return (
    <a
      href={download.url}
      target="_blank"
      rel="noopener noreferrer"
      className={`
        inline-flex items-center justify-center border rounded-lg font-medium transition-all duration-200
        hover:scale-105 hover:shadow-lg focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-green-500
        ${sizeStyles[size]} ${variantStyles[variant]}
      `}
    >
      <span className="text-2xl mr-3">{download.icon}</span>
      <div className="text-left">
        <div className="font-semibold">{download.label}</div>
        {size !== "sm" && (
          <div className="text-xs opacity-75">{download.subtitle}</div>
        )}
      </div>
    </a>
  );
};

export const AppDownload = ({ platform = "both", variant = "default", layout = "horizontal" }) => {
  const platforms = platform === "both" ? ["ios", "android"] : [platform];

  if (layout === "vertical") {
    return (
      <div className="flex flex-col space-y-4">
        {platforms.map((p) => (
          <DownloadButton key={p} platform={p} variant={variant} size="lg" />
        ))}
      </div>
    );
  }

  return (
    <div className="flex flex-col sm:flex-row gap-4">
      {platforms.map((p) => (
        <DownloadButton key={p} platform={p} variant={variant} />
      ))}
    </div>
  );
};
